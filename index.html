Simple View 
        
        // Backbone View
        // updated independently when the model changes
        var CalendarItemView = Backbone.View.extend({

            id : 'calendar_item',

            template : Handlebars.compile( $( '#calendar_item' ).html() ),

            events : {
                'click #item_cancel' : function() {
                    history.back();
                }
            },
            initialize : function() {
                this.model.bind( 'change', this.render );
                this.render();
            },
            render : function() {
                $( this.el ).html( this.template( this.model.toJSON() ) );
            }
        });
